DS_LIST_ReversingLinkedList

这题保守估计写了6个小时TAT

Description

Given a constant K and a singly linked list L, you are supposed to reverse the links of every K elements on L. For example, given L being 1→2→3→4→5→6, if K=3, then you must output 3→2→1→6→5→4; if K=4, you must output 4→3→2→1→5→6.

Input Specification:

Each input file contains one test case. For each case, the first line contains the address of the first node, a positive N (≤105) which is the total number of nodes, and a positive K (≤N) which is the length of the sublist to be reversed. The address of a node is a 5-digit nonnegative integer, and NULL is represented by -1.

Then N lines follow, each describes a node in the format:

Address Data Next

where Address is the position of the node, Data is an integer, and Next is the position of the next node.

Output Specification:

For each case, output the resulting ordered linked list. Each node occupies a line, and is printed in the same format as in the input.

Sample Input:

00100 6 4
00000 4 99999
00100 1 12309
68237 6 -1
33218 3 00000
99999 5 68237
12309 2 33218

Sample Output:

00000 4 33218
33218 3 12309
12309 2 00100
00100 1 99999
99999 5 68237
68237 6 -1

Source Code

//Reversing Linked List
//special case
#include<stdio.h>
#include<stdlib.h>
#define N 100001

struct Node {
    int Address;
    int Data;
    int Next;
};

void Rev(struct Node* l, struct Node* rl, int n, int ReUnit, int Head);
void Print(struct  Node* rl, int n);
int Check(struct Node* l, int Head);
int main()
{
    int n, ReUnit, Head, i, addr, data, next;
    scanf("%d %d %d", &Head, &n, &ReUnit);
    struct Node l[N];
    struct Node rl[n];
    int inList[N] = {0};
    inList[Head] = 1;
    for(i = 0; i < n; i++)
    {
        scanf("%d", &addr);
        scanf("%d", &data);
        scanf("%d", &next);
        l[addr].Address = addr;
        l[addr].Data = data;
        l[addr].Next = next;
        if(next != -1)    inList[next] = 1;
    }
    n = Check(l, Head);
    //printf("%d\n",n);
    Rev(l, rl, n, ReUnit, Head);
    Print(rl, n);
    system ("pause");
    return 0;
}

int Check(struct Node* l, int Head)
{
    int num=0;
    int index = Head;
    while(index != -1)
    {
        num++;
        index = l[index].Next;
    }
    return num;
}

void Rev(struct Node* l, struct Node* rl, int n, int ReUnit, int Head)
{
    int times = n/ReUnit;
    int i, nehead=0, rindex = 0, j, k=0, haddr, index;
    haddr = l[Head].Address;
    if(ReUnit == 1 || times == 0)
    {
        index = haddr;
        rindex = 0;
        while(index != -1)
        {
            rl[rindex].Address = l[index].Address;
            rl[rindex].Data = l[index].Data;
            rl[rindex].Next = l[index].Next;
            rindex++;
            index = l[index].Next;
        }
    }
    else 
    {
        nehead = haddr;
        for(i = 1; i <= times; i++)
        {
            rindex = i*ReUnit - 1;
            index = nehead;
            for(j = 0; j < ReUnit-1; j++)
            {
                rl[rindex].Address = l[index].Address;
                rl[rindex].Data = l[index].Data;    
                rindex--;
                index = l[index].Next;
            }
            nehead = l[index].Next;
            rl[rindex].Address = l[index].Address;
            rl[rindex].Data = l[index].Data;
        }
        rindex += ReUnit;
        if(rindex < n )
        {
            index = nehead;
            for(; index != -1; index = l[index].Next)
            {
                rl[rindex].Address = l[index].Address;
                rl[rindex].Data = l[index].Data;
                rindex++;
            }
        }

        for( k = 0; k < n-1; k++)
        {
            rl[k].Next = rl[k+1].Address;
        }
        rl[k].Next = -1;
    }


    return ;
}
void Print(struct Node* rl, int n)
{
    int i;
    for(i = 0; i < n-1; i++)
    {
        printf("%05d %d %05d\n", rl[i].Address, rl[i].Data, rl[i].Next);
    }
    printf("%05d %d %d", rl[i].Address, rl[i].Data, rl[i].Next);
    return ;
}
