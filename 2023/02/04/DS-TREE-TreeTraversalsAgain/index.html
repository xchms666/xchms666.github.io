DS_TREE_TreeTraversalsAgain

2h 其实是之前的作业题，之前尝试写过没写出来

这次只debug了一次还是蛮开心的

Description

An inorder binary tree traversal can be implemented in a non-recursive way with a stack. For example, suppose that when a 6-node binary tree (with the keys numbered from 1 to 6) is traversed, the stack operations are: push(1); push(2); push(3); pop(); pop(); push(4); pop(); pop(); push(5); push(6); pop(); pop(). Then a unique binary tree (shown in Figure 1) can be generated from this sequence of operations. Your task is to give the postorder traversal sequence of this tree.

Figure 1

Input Specification:

Each input file contains one test case. For each case, the first line contains a positive integer N (≤30) which is the total number of nodes in a tree (and hence the nodes are numbered from 1 to N). Then 2N lines follow, each describes a stack operation in the format: "Push X" where X is the index of the node being pushed onto the stack; or "Pop" meaning to pop one node from the stack.

Output Specification:

For each test case, print the postorder traversal sequence of the corresponding tree in one line. A solution is guaranteed to exist. All the numbers must be separated by exactly one space, and there must be no extra space at the end of the line.

Sample Input:

6
Push 1
Push 2
Push 3
Pop
Pop
Push 4
Pop
Pop
Push 5
Push 6
Pop
Pop

Sample Output:

3 4 2 6 5 1

Ideas

回顾《数据结构》课程“树的遍历”一节，我们知道树的前中后三中遍历方式其实访问结点的顺序是一样的，只不过打印结点的顺序不同

图源浙江大学MOOC课程《数据结构》

因此我们知道，push的顺序其实就是先序遍历的顺序，因为push是第一次遇到结点

所以题目变成：已知树的先序和中序遍历，求后序遍历

一般思路下，本题关键步骤就是两步，一根据先序中建树，二后序遍历

如果递归的思维比较强，还可以直接得出后序遍历序列

建树

回顾课程笔记，我们知道，根据中序遍历结果和先/后序遍历结果可以唯一确定一棵二叉树

图源浙江大学MOOC课程《数据结构》

先序遍历序列中，对于每一个结点，它的遍历顺序一定是“该节点，左儿子，右儿子”。因此，先序遍历序列中，对于每一个结点，其都在其子树的左边；对于整棵树，最左一个元素必然是树的根节点。

中序遍历序列中，对于每一个结点，它的遍历顺序一定是“左儿子，该节点，右儿子”。因此，中序遍历序列中，对于每一个结点，左子树在其左边，右子树在其右边；对于整棵树，中序遍历序列的根节点的左边必然是左子树，右边必然是右子树。

因此，建树的思路便是：先序遍历序列的最左边是子树的根节点，在中序遍历序列中找到该结点，分出左右子树，再以同样方式递归建树

直接

直接求出后序遍历序列其实也很简单

后序遍历序列的特征不再赘述。对于每一个子树的遍历序列，先序序列的最左边元素是后序序列的最右元素，即根结点可以直接确定。再递归通过中序确定左右子树即可。

在这里递归的思维真的很美妙，我们只要把每个结点都看作一个二叉子树中的根结点，只关注根结点在三种遍历方式中的位置关系就可以递归得到后序遍历序列了。

//Tree Traversals Again
#include<stdio.h>
#include<stdlib.h>
#define MaxSize 30
int pre[MaxSize], in[MaxSize], post[MaxSize];
int pi = 0;
int ii = 0;
int NodeNum = 0;
typedef struct node{
    int data;
    struct node* left;
    struct node* right;
} Tnode;
typedef Tnode* root;
int iStack[MaxSize];
int itop = -1;
void push(int t);
int pop();
int Find(int tar);
void Read();
root BuildTree(int pl, int pr, int il, int ir);
void PostOrderTraversal(root Root);
int main()
{
    root Root = (root)malloc(sizeof(struct node));
    Read();
    pi = ii = 0;
    Root = BuildTree(0,NodeNum-1,0,NodeNum-1);
    pi = 0;
    PostOrderTraversal(Root);
    printf("%d", post[0]);
    for(pi = 1; pi < NodeNum; pi++) printf(" %d", post[pi]);
    system("pause");
    return 0;
}
void Read()
{
    scanf("%d\n", &NodeNum);
    int i = 2*NodeNum;
    for(;i > 0; i--)
    {
        getchar();
        char c;
        int tmp;
        c = getchar();
        if(c == 'u')
        {
            getchar();getchar();getchar();
            scanf("%d\n", &tmp);
            pre[pi++] = tmp;
            push(tmp);
        }
        else
        {
            getchar();getchar();
            in[ii++] = pop();
        }
    }
    return ;
}
root BuildTree(int pl, int pr, int il, int ir)
{
    int iindex = 0;
    root tmp = (root)malloc(sizeof(struct node));
    int rd = pre[pl];
    tmp->data = rd;
    tmp->left = NULL;
    tmp->right = NULL;
    iindex = Find(rd);
    int llen = iindex-il;
    int rlen = ir - iindex;
    if(llen > 0)    tmp->left = BuildTree(pl+1, pl+llen, il, iindex-1);
    if(rlen > 0)    tmp->right = BuildTree(pl+llen+1,pr,iindex+1,ir);
    return tmp;
}
void PostOrderTraversal(root Root)
{
    if(Root){
        PostOrderTraversal(Root->left);
        PostOrderTraversal(Root->right);
        post[pi++] = Root->data;
    }
    return ;
}
void push(int t)
{
    iStack[++itop] = t;
    return ;
}
int pop()
{
    return iStack[itop--];
}
int Find(int tar)
{
    int index = 0;
    while(in[index++]!=tar);
    return --index;
}

Version-3

没有建树，直接求出后序遍历序列

//Tree Traversals Again-3
#include<stdio.h>
#include<stdlib.h>
#define MaxSize 30

int pre[MaxSize], in[MaxSize], post[MaxSize];
int poi = 0;
int NodeNum = 0;
int Stack[MaxSize];
int top = -1;

void push(int t);
int pop();
int Find(int tar,int l,int r);
void Read();
void Solve(int prel, int inl, int pol, int len);

int main()
{
    int i;
    Read();
    Solve(0,0,0,NodeNum);
    printf("%d", post[0]);
    for(i = 1; i < NodeNum; i++)    printf(" %d", post[i]);
    system("pause");
    return 0;
}
void Read()
{
    int pri = 0;
    int ii = 0;
    scanf("%d\n", &NodeNum);
    int i = 2*NodeNum;
    for(;i > 0; i--)
    {
        getchar();
        char c;
        int tmp;
        c = getchar();
        if(c == 'u')
        {
            getchar();getchar();getchar();
            scanf("%d\n", &tmp);
            pre[pri++] = tmp;
            push(tmp);
        }
        else
        {
            getchar();getchar();
            in[ii++] = pop();
        }
    }
    return ;
}

void push(int t)
{
    Stack[++top] = t;
    return ;
}
int pop()
{
    return Stack[top--];
}
int Find(int tar, int l, int r)
{
    int i = l-1;
    while(in[++i]!=tar && i<r);
    return i;
}
void Solve(int prel, int inl, int pol, int len)
{
    if(len == 0)    return ;
    if(len == 1)
    {
        post[pol] = pre[prel];
        return ;
    }
    int ln,rn,rix;
    post[pol+len-1] = pre[prel];
    rix = Find(pre[prel],inl,inl+len-1);
    ln = rix-inl;
    rn = inl+len-1-rix;
    Solve(prel+1,inl,pol,ln);
    Solve(prel+ln+1,rix+1,pol+ln,rn);
    return ;
}
