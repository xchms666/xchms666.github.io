Max Subsequence Sum

1. O(N^3^)

//1
int MaxSubsequenceSum(const int A[ ],  int  N )
{
    int max = 0;
    int i,j,k;
    int cur = 0;
    for(i = 0; i < N; i++)
    {
        for(j = i; j < N; j++)
        {
            cur = 0;
            for(k = i; k < j; k++)
            {
                cur += A[k];
                max = Max(cur, max, 0);
            }
        }
    }

    return max;
}

2. O(N^2^)

3. O(NlogN)

//3
int MaxSubsequenceSum(const int A[ ],  int  li, int ri )
{
    int max = 0;
    if(li == ri )
    {
        if(A[li] < 0)    return 0;
        else    return A[li];
    }
    int mi = (li + ri)/2;    
    int lm = 0,mm = 0,rm = 0;
    lm = MaxSubsequenceSum(A,li,mi);
    rm = MaxSubsequenceSum(A,mi+1,ri);
    int lbm = 0, rbm = 0, bm = 0;
    int i,j;
    int cur = 0;
    for(i = mi; i >= li; i--)
    {
        cur += A[i];
        lbm = Max(lbm, cur);
    }
    cur = 0;
    for(i = mi+1; i<=ri; i++)
    {
        cur += A[i];
        rbm = Max(rbm, cur);
    }
    bm = lbm+rbm;
    max = Max(Max(lm,rm),bm);
    return max;
}

4. O(N)

//4
int MaxSubsequenceSum(const int A[ ],  int  N )
{
    int max = 0;
    int i,cur = 0;
    for(i = 0; i < N; i++)
    {
        cur += A[i];
        if(cur < 0)    cur = 0;
        max = Max(max,cur);
    }

    return max;
}

Linked List

定义

typedef struct LNode *PtrToLNode
struct LNode{
    ElementType Data;
    PtrToLNode Next;
}
typedef PtrToLNode List;
typedef PtrToLNode Position;

查找

Position Find( List L, ElementType X )
{
    Position p = L;
    while(p && p->Data!=X)    p = p->Next;
    return p;
}

带头节点的插入

bool Insert( List L, ElementType X, Position P )
{
    Position pre = L;
    while(pre && pre->Next != P)    pre = pre->Next;
    if(pre)
    {
        Position tmp = (Position)malloc(sizeof(struct LNode));
        tmp->Next = P;
        tmp->Data = X;
        pre->Next = tmp;
        return true;
    }
    printf("输入参数错误\n");
    return false;
}

带头结点的删除

bool Delete( List L, Position P )
{
    Position pre = L;
    while(pre && pre->Next != P)    pre = pre->Next;
    if(pre && P)
    {
        pre->Next = P->Next;
        free(P);
        return true;
    }
    else
    {
        printf("删除位置参数错误\n");
        return false;
    }
}
